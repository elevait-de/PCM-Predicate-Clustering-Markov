SELECT DISTINCT  ?artist ?artistName ?occupation ?shortDescription ?birthdate ?summary ?photo ?website
WHERE
  { ?artist <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> dbpedia:Artist .
    ?artist rdfs:label ?artistName
    OPTIONAL
      { ?artist dbpedia:birthDate ?birthdate }
    OPTIONAL
      { ?artist dbprop:occupation ?occupation
        FILTER isLiteral(?occupation)
      }
    OPTIONAL
      { ?artist dbprop:shortDescription ?shortDescription }
    OPTIONAL
      { ?artist dbpedia:thumbnail ?photo }
    OPTIONAL
      { ?artist dbpedia:abstract ?summary
        FILTER ( lang(?summary) = "fr" )
      }
    OPTIONAL
      { ?artist foaf:page ?website }
    FILTER regex(?artistName, "gérard depardieu", "i")
    FILTER ( lang(?artistName) = "en" )
  }

SELECT  *
WHERE
  { ?artist rdf:type dbpo:MusicalArtist .
    ?artist rdfs:label ?name
    FILTER regex(?artist, "White", "i")
  }

SELECT DISTINCT  ?subject ?predicate ?object
WHERE
  {   { <http://dbpedia.org/resource/John_Lennon> ?predicate ?object }
    UNION
      { ?subject ?predicate <http://dbpedia.org/resource/John_Lennon> }
    FILTER ( ( lang(?object) = "en" ) || ( lang(?object) = "" ) )
  }
ORDER BY ?predicate ?subject
OFFSET  200
LIMIT   200

PREFIX  geo:  <http://www.w3.org/2003/01/geo/wgs84_pos#>
PREFIX  p:    <http://dbpedia.org/property/>

SELECT  *
WHERE
  { ?player p:currentclub <http://dbpedia.org/resource/Arsenal_F.C.>
    OPTIONAL
      { ?player p:cityofbirth ?city }
    OPTIONAL
      { ?player p:dateOfBirth ?dob }
    OPTIONAL
      { ?player p:clubnumber ?no }
    OPTIONAL
      { ?player p:position ?position }
    OPTIONAL
      { ?player p:image ?image }
    OPTIONAL
      {   { ?city geo:long ?long }
        UNION
          { ?city p:redirect ?city2 .
            ?city2 geo:long ?long
          }
      }
    OPTIONAL
      {   { ?city geo:lat ?lat }
        UNION
          { ?city p:redirect ?city3 .
            ?city3 geo:lat ?lat
          }
      }
  }
